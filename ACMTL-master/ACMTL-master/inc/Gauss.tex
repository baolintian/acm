void Gauss()
{
    for(int i=1;i<=tot+1;i++)
    {   
        int tmp=i;
        for(int j=i;j<=tot+1;j++)
            if(fabs(A[j][i])>fabs(A[tmp][i])) tmp=j;
        for(int j=1;j<=tot+2;j++) swap(A[i][j],A[tmp][j]);
        for(int j=i+1;j<=tot+1;j++)
        {   
            double o=A[j][i]/A[i][i];
            for(int k=1;k<=tot+2;k++)
                A[j][k]-=o*A[i][k];
        }    
    }   
}
